    4. Realiza un servidor de sockets UDP que imprima por pantalla si el cliente se ha conectado (1P).
      5. 
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
                        
public class UDPServidor {
                        
    private static final int PUERTO = 12345;
                        
    public static void main(String[] args) {
        try (DatagramSocket socket = new DatagramSocket(PUERTO)) {
            byte[] buffer = new byte[1024];
                        
            System.out.println("Servidor UDP escuchando en el puerto " + PUERTO);
                        
            while (true) {
                DatagramPacket paquete = new DatagramPacket(buffer, buffer.length);
                socket.receive(paquete);
                        
                InetAddress direccionCliente = paquete.getAddress();
                int puertoCliente = paquete.getPort();
                        
                System.out.println("Cliente conectado desde " + direccionCliente.getHostAddress() + ":" + puertoCliente);
            }
        } catch (Exception e) {
            System.err.println("Error: " + e.getMessage());
        }
    }
}
